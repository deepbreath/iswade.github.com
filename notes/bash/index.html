    安装软件

    # ubuntu 软件安装
apt-get install SOFTWARE  # 安装软件
apt-get update            # 更新源之后，用于更新索引
apt-get upgrade           # 升级软件
do-release-upgrade        # 升级操作系统

# yum 软件安装
yum install SOFTWARE

# python 软件安装
pip install MODULE

    系统管理

    # 设置用户名
hostname -b NEWHOSTNAME  # 修改的是 /proc/ 文件系统中 hostname 中的名称
/etc/hostname            # 设置永久用户名

# sudo 权限设置
/etc/sudoers             # 对于 sudo 权限，不能执行 buildin 命令

# 定时任务
crontab -l               # 查看定时任务 
crontab -e               # 编辑定时任务

cd .ssh
cat id_rsa.pub > authorized_keys
chmod 600 authorized_keys 
exit
