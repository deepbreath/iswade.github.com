    Arrays

    在 Go 中，数组是指定长度的元素的一个序列。

    package main

import "fmt"

func main() {

   // 创建了一个有5个元素的数组
   // 元素的类型和长度是数组的类型
   // 默认情况下被赋值为“零”
    var a [5]int
    fmt.Println("emp:", a)

   // 可以适用索引来赋值
   // 也可以通过索引来获取值
    a[4] = 100
    fmt.Println("set:", a)
    fmt.Println("get:", a[4])

   // 内置函数 len 用来返回数组的长度
    fmt.Println("len:", len(a))

   // 使用这种语法在一行中声明和初始化数组
    b := [5]int{1, 2, 3, 4, 5}
    fmt.Println("dcl:", b)

   // 数组类型是一维的，但可以组合类型来构造多维数据结构
    var twoD [2][3]int
    for i := 0; i < 2; i++ {
        for j := 0; j < 3; j++ {
            twoD[i][j] = i + j
        }
    }
    fmt.Println("2d: ", twoD)
}

    输出结果：

    $ go run arrays.go
emp: [0 0 0 0 0]
set: [0 0 0 0 100]
get: 100
len: 5
dcl: [1 2 3 4 5]
2d:  [[0 1 2] [1 2 3]]

    用 fmt.Println 函数输出数组的时候，以[v1 v2 v3 ...]这种形式输出。
