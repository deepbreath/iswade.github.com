    Constants

    Go 支持字符、字符串、布尔和数字常量。

    // Go supports _constants_ of character, string, boolean,
// and numeric values.

package main

import "fmt"
import "math"

// const 用来定义常量
const s string = "constant"

func main() {
    fmt.Println(s)

    // const 语句可以出现在任何 var 语句可以出现的地方
    const n = 500000000

    // 常量表达式会以任意精度执行计算
    const d = 3e20 / n
    fmt.Println(d)

    // 数字常量没有类型除非给的一个类型或者通过显示转换
    fmt.Println(int64(d))

    // 一个数字可以给定类型，在使用它的上下文中，例如一个变量赋值或者函数调用
    // 例如，这里 math.Sin 期望使用 float64 类型
    fmt.Println(math.Sin(n))
}

    $ go run constant.go
