    Errors

    在 Go 语言中，通常都是通过显式分离返回值得方式处理错误码。 这与 Java 和 Ruby 等语言使用的异常，以及 C 语言中有时使用的重载单个结果/错误码不同。 Go 中以更简单的形式来查看哪个函数返回错误，然后用相同的语言结构处理。

    package main

import (
    "errors"
    "fmt"
)

// 按照惯例，错误码是最后一个返回值，并且有类型为内置接口 error
func f1(arg int) (int, error) {
    if arg == 42 {
      // errors.New 根据给定的错误信息构造一个基本的 error 值
        return -1, errors.New("can't work with 42")
    }

   // nil 值表示没有错误
    return arg + 3, nil
}

// 通过实现 Error() 方法，可以将定制类型作为 errors
// 这是上面例子的变体，使用定制类型来显示表示参数错误
type argError struct {
    arg  int
    prob string
}

func (e *argError) Error() string {
    return fmt.Sprintf("%d - %s", e.arg, e.prob)
}

func f2(arg int) (int, error) {
    if arg == 42 {
      // 这种情况下，使用 &argError 语法构建一个新的结构
      // 提供两个字段 arg 和 prob 作为取值
        return -1, &argError{arg, "can't work with it"}
    }

    return arg + 3, nil
}

func main() {
   // 下面两个循环，测试了每个错误返回函数
   // if 行内检查错误码在 Go 中是比较常见的做法
    for _, i := range []int{7, 42} {
        if r, e := f1(i); e != nil {
            fmt.Println("f1 failed:", e)
        } else {
            fmt.Println("f1 worked:", r)
        }
    }

    for _, i := range []int{7, 42} {
        if r, e := f2(i); e != nil {
            fmt.Println("f2 failed:", e)
        } else {
            fmt.Println("f2 workded:", r)
        }
    }

   // 如果想使用定制错误的数据，需要通过类型断言作为定制错误类型的实例来获得错误。
    _, e := f2(42)
    if ae, ok := e.(*argError); ok {
        fmt.Println(ae.arg)
        fmt.Println(ae.prob)
    }
}

    运行结果：

    $ go run errors.go
f1 worked: 10
f1 failed: can't work with 42
f2 worked: 10
f2 failed: 42 - can't work with it
42
can't work with it
