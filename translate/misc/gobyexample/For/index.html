    For

    for 是 Go 语言中为唯一的一个循环结构，for 循环有三个基本的类型。

    package main

import "fmt"

func main() {
    i := 1

   // 最基本的类型，仅有一个条件
    for i <= 3 {
        fmt.Println(i)
        i = i + 1
    }

   // 经典的 初始化/条件/执行后操作
    for j := 7; j <= 9; j++ {
        fmt.Println(j)
    }

   // for 循环没有条件，无限循环，直到 break 或者 return 返回
    for {
        fmt.Println("loop")
        break
    }

   // 也可以使用 continue 进入到下一次的循环中
    for n := 0; n <= 5; n++ {
        if n%2 == 0 {
            continue
        }
        fmt.Println(n)
    }
}
