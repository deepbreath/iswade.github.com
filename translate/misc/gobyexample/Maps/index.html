    Maps

    Maps 是 Go 内置的关联数据类型（在其它语言中有时被叫做 hashes 或者 dicts）。

    package main

import (
    "fmt"
)

func main() {
    // 可以用 make(map[key-type]val-type) 创建一个空的 map
    m := make(map[string]int)

    // 通过典型的语法 name[key] = value 设置键值对 
    m["k1"] = 7
    m["k2"] = 13

    // 用 Println 可以打印出所有的键值对
    fmt.Println("map:", m)

    // 用 name[key] 获得一个键的值
    v1 := m["k1"]
    fmt.Println("v1: ", v1)

    // 内置的 len 可以获得键值对的数量
    fmt.Println("len:", len(m))

    // 内置的 delete 可以用来删除键值对
    delete(m, "k2")
    fmt.Println("map:", m)

    // 可选的第二个返回值可以用来表示 key 是否在 map 中
    // 可以用来区分是不存在的 key，还是零值 key
    // 这里我们不需要 value 本身，所以用空白标示 _ 来忽略
    _, prs := m["k2"]
    fmt.Println("prs", prs)

    // 可以在一行中声明和初始化一个新的 map
    n := map[string]int{"foo": 1, "bar": 2}
    fmt.Println("map:", n)
}

    运行结果：

    $ go run maps.go 
map: map[k1:7 k2:13]
v1:  7
len: 2
map: map[k1:7]
prs: false
map: map[foo:1 bar:2]

    需要注意的是，用 fmt.Println 函数打印的格式 map[k:v k:v]。
