    Methods

    Go 语言支持在结构类型上定义方法。

    package main

import (
    "fmt"
)

type rect struct {
    width, height int
}

// area 方法有一个接收器类型 *rect
func (r *rect) area() int {
    return r.width * r.height
}

// 方法可以被定义为指针或者值得接收器类型，这个是值接收器的例子
func (r rect) perim() int {
    return 2*r.width + 2*r.height
}

// 可以用来修改结构中字段的内容
func (r *rect) mutate() {
    r.width = 20
    r.height = 10
}

func main() {
    r := rect{width: 10, height: 5}

   // 这里我们调用我们定义的结构的两个方法
    fmt.Println("area:", r.area())
    fmt.Println("perim:", r.perim())

   // 对于方法调用，Go 自动处理值和指针的转换
   // 可以用指针接收器类型避免方法调用时的拷贝，这样就可以修改接受结构
    rp := &r
    fmt.Println("area:", rp.area())
    fmt.Println("perim:", rp.perim())

    rp.mutate()
    fmt.Println("area:", rp.area())
    fmt.Println("perim:", rp.perim())
}

    运行结果：

    ￥ go run methods.go
area: 50
perim: 30
area: 50
perim: 30
area: 200
perim: 60
