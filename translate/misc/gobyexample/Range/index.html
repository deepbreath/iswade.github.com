    Range

    Range 可以在一些数据结构上进行单个元素的迭代。 下面我们来看下如何进行在我们已经学习过的一些数据结构上进行 range 操作。

    package main

import (
    "fmt"
)

func main() {

    // 通过 range 求 slice 的所有数字的和，数组也一样
    nums := []int{2, 3, 4}
    sum := 0
    for _, num := range nums {
        sum += num
    }
    fmt.Println("sum:", sum)

    // 通过 range 获得数组和切片的 index 和 value
    // 上面我们不需要 index，所以用空白标识符_替代
    // 但实际上有时候我们需要 index
    for i, num := range nums {
        if num == 3 {
            fmt.Println("index:", i)
        }
    }

    // map 的 range 可以用于迭代键值对
    kvs := map[string]string{"a": "apple", "b": "banana"}
    for k, v := range kvs {
        fmt.Printf("%s - %s\n", k, v)
    }

    // range 也可以仅用于迭代 key 
    for k := range kvs {
        fmt.Println("key:", k)
    }

    // 字符串上使用 range，用于迭代 Unicode 码
    // 第一个值是 rune 的起始字节的 index
    // 第一个值是 rune  自己
    for i, c := range "go" {
        fmt.Printf("%d %c\n", i, c)
        fmt.Println(i, c)
    }
}

    运行结果：

    $ go run range.go
sum: 9
index: 1
a -> apple
b -> banana
key: a
key: b
0 103
1 111
