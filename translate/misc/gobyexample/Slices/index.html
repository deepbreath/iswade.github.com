    Slices

    切片是 Go 语言中比较重要的一种类型，对于序列给出了一种比数组更加强大的接口。

    package main

import "fmt"

func main() {

   // 与数组不同，切片仅以包含的元素为类型（不包含元素的个数）
   // 创建空的非零长度的切片，可以适用 make 内置函数
   // 这里创建了一个长度为3的字符串切片（初始化为零值）
    s := make([]string, 3)
    fmt.Println("emp:", s)

   // 可以像数组一样设置或者获取值
    s[0] = "a"
    s[1] = "b"
    s[2] = "c"
    fmt.Println("set:", s)
    fmt.Println("get:", s[2])

   // len 返回切片的长度
    fmt.Println("len:", len(s))

   // 除了基本操作，切片还支持一些其它比数组更丰富的操作
   // 一个是内置函数 append，会返回一个包含一个或多个新值的切片
   // 注意：我们需要接受 append 操作的返回值，因为我们会得到一个新切片的值
    s = append(s, "d")
    s = append(s, "e", "f")
    fmt.Println("apd:", s)

   // 切片可以通过 copy 进行拷贝
   // 这里我们创建一个空的切片 c 跟 s 的长度一致，然后拷贝 s 到 c
    c := make([]string, len(s))
    copy(c, s)
    fmt.Println("cpy:", c)

   // 切片支持 "slice" 操作符，语法格式 slice[low:high]
   // 例子中，获得了元素 s[2], s[3] 和 s[4]三个元素的切片
    l := s[2:5]
    fmt.Println("sl1:", l)

   // 切片可以到 s[5]，但不包括 s[5]
    l = s[:5]
    fmt.Println("sl2:", l)

   // 可以从 s[2]（包含）开始，然后到后面的所有元素
    l = s[2:]
    fmt.Println("sl3:", l)

   // 可以定义和初始化一个切片变量在一行中
    t := []string{"g", "h", "i"}
    fmt.Println("dcl:", t)

   // 切片可以组成多维数据结构，内部切片的长度可变，跟多维数组不同
    twoD := make([][]int, 3)
    for i := 0; i < 3; i++ {
        innerLen := i + 1
        twoD[i] = make([]int, innerLen)
        for j := 0; j < innerLen; j++ {
            twoD[i][j] = i + j
        }
    }

    fmt.Println("2d: ", twoD)
}

    运行结果：

    $ go run slices.go

len: 3
apd: [a b c d e f]

sl1: [c d e]
sl2: [a b c d e]
sl3: [c d e f]
dcl: [g h i]
2d:  [[0] [1 2] [2 3 4]]
